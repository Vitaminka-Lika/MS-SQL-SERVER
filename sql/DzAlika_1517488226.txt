use HomeWork
declare @c int=0, @g int=0, @z int
select @g=min(id) from tmp
select @c=count(1) from tmp
while (@g<@c)
begin
select @z=id from tmp where id=@g
if(@z%2=0)
begin
DELETE FROM tmp
where id=@g
end
set @g=@g+1
end
